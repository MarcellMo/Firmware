1function xdot_s = OC_fx(x,mass)
 2% state part of state derivatives xdot = f(x)+g(x)*u
 3% Author: Bernd Messnarz, Last Revision: 2015|03|22
 4% definition of states
 5% x =[u; v; w; x; y; z; p; q; r; phi; theta; psi]
 6  VKb = x(1:3);  % Kinematic velocity in body frame
 7  OKb = x(7:9);  % Angular velocity of the body in body frame
 8  
 9% Initialization
10  xdot_s  = zeros(6,1);
11
12% DCM to transform wind into body axes              
13  sphi = sin(x(10)); cphi = cos(x(10));
14  sthe = sin(x(11)); cthe = cos(x(11));
15  
16  if abs(cthe) <= single(0.00001)
17    cthe = single(0.00001) * sign(cthe);
18  end
19  
20  tthe = sthe/cthe; 
21
22% Body drag force and moment (damping)
23  VAb = VKb;   % for drag force
24
25  % Drag force
26  FDb = -0.062 * norm(VAb) .* VAb; % cD = 1.24*1*0.1*1/2  (rho*CD*A*0.5) (=0.062)
27 
28% Gravity in body frame
29
30  gb =  9.80665 *[-sthe; sphi*cthe; cphi*cthe];
31  
32% EOM
33% H_eulerdot_omega
34  Hedom = [1  sphi*tthe   cphi*tthe;
35           0  cphi       -sphi;
36           0  sphi/cthe   cphi/cthe];
37
38% Force (drag, gravity and inertia)
39  xdot_s(1:3)   = gb + FDb/mass -  cross(OKb, VKb); 
40% Moment (drag, gravity and inertia)
41% Attitude
42  xdot_s(4:6) = Hedom * OKb;
